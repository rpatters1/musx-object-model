68class ElementLinker {
69public:
77 using Resolver = std::function<void(const dom::DocumentPtr&)>;
78
88 void addResolver(Resolver resolver, const std::string_view& key = {})
89 {
90 if (!key.empty()) {
91 if (registeredResolvers.count(key) > 0) {
92 return; // Skip adding if the resolver with the same key is already registered
93 }
94 registeredResolvers.insert(key);
95 }
96 resolvers.emplace_back(std::move(resolver));
97 }
98
108 void resolveAll(const dom::DocumentPtr& document)
109 {
110 for (auto& resolver : resolvers) {
111 resolver(document);
112 }
113 resolvers.clear();
114 registeredResolvers.clear();
115 }
116
117private:
119 std::vector<Resolver> resolvers;
120
126 std::unordered_set<std::string_view> registeredResolvers;
127};

264template <typename T>
265struct FieldPopulator : public FactoryBase
266{
267 static void populate(const std::shared_ptr<T>& instance, const XmlElementPtr& element)
268 {
269 for (auto child = element->getFirstChildElement(); child; child = child->getNextSibling()) {
270 auto it = elementXref().find(child->getTagName());
271 if (it != elementXref().end()) {
272 std::get<1>(*it)(child, instance);
273 } else {
274 const bool requireFields = [instance]() {
275 if constexpr (std::is_base_of_v<Base, T>) {
276 return instance->requireAllFields();
277 } else {
278 return true;
279 }
280 }();
281 if (requireFields) {
282 std::string msg = "xml element <" + element->getTagName() + "> has child <" + child->getTagName() + "> which is not in the element list.";
283#ifdef MUSX_THROW_ON_UNKNOWN_XML
284 throw unknown_xml_error(msg);
285#else
286 util::Logger::log(util::Logger::LogLevel::Warning, msg);
287#endif
288 }
289 }
290 }
291 if constexpr (std::is_base_of_v<Base, T>) {
292 instance->integrityCheck();
293 }
294 }
295
296 static void populate(const std::shared_ptr<T>& instance, const XmlElementPtr& element, ElementLinker& elementLinker)
297 {
298 populate(instance, element);
299 for (const auto& resolver : ResolverArray<T>::value) {
300 elementLinker.addResolver(resolver);
301 }
302 }
303
304 template <typename... Args>
305 static std::shared_ptr<T> createAndPopulate(const XmlElementPtr& element, Args&&... args)
306 {
307 return FieldPopulator<T>::createAndPopulateImpl(element, std::forward<Args>(args)...);
308 }
309
310private:
311 static const std::unordered_map<std::string_view, XmlElementPopulator<T>>& elementXref()
312 {
313 static const std::unordered_map<std::string_view, XmlElementPopulator<T>> xref = []()
314 {
315 std::unordered_map<std::string_view, XmlElementPopulator<T>> retval;
316 for (std::size_t i = 0; i < T::XmlMappingArray.size(); i++) {
317 const XmlElementDescriptor<T> descriptor = T::XmlMappingArray[i];
318 retval[std::get<0>(descriptor)] = std::get<1>(descriptor);
319 }
320 return retval;
321 }();
322 return xref;
323 }
324
325 template <typename... Args>
326 static std::shared_ptr<T> createAndPopulateImpl(const XmlElementPtr& element, Args&&... args)
327 {
328 auto instance = std::make_shared<T>(std::forward<Args>(args)...);
329 FieldPopulator<T>::populate(instance, element);
330 return instance;
331 }
332};
