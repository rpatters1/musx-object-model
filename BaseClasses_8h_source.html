95{
96public:

113 virtual ~Base() noexcept(false) = default;
114
120 DocumentPtr getDocument() const
121 {

123 assert(document); // program bug if this pointer goes out of scope.
124 return document;
125 }
126
130 Cmper getPartId() const { return m_partId; }
131
135 ShareMode getShareMode() const { return m_shareMode; }

146 {
147 m_unlinkedNodes.insert(nodeName);
148 }
149
157 virtual void integrityCheck() const { }
158
169 virtual bool requireAllFields() const { return true; }
170
171protected:
179 Base(const DocumentWeakPtr& document, Cmper partId, ShareMode shareMode)
180 : m_document(document), m_partId(partId), m_shareMode(shareMode) {}
181
183 Base& operator=(const Base&) { return *this; }
184
185private:
186 const DocumentWeakPtr m_document;
187 const Cmper m_partId;
188 const ShareMode m_shareMode;
189 SharedNodes m_unlinkedNodes;
