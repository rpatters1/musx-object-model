106class Base
107{
108public:
110 using SharedNodes = std::set<std::string>;
111
114 enum class ShareMode
115 {
116 // this enum was giving Doxygen fits until I switched to using the slash-splat-bang-lt comment commands.
117 All,
118 Partial,
119 None
120 };
121
125 virtual ~Base() noexcept(false) = default;
126
132 DocumentPtr getDocument() const
133 {
134 auto document = m_document.lock();
135 assert(document); // program bug if this pointer goes out of scope.
136 return document;
137 }
138
142 Cmper getPartId() const { return m_partId; }
143
145 std::shared_ptr<others::PartDefinition> getPartDefinition() const;
146
150 ShareMode getShareMode() const { return m_shareMode; }
151
155 const SharedNodes& getUnlinkedNodes() const { return m_unlinkedNodes; }
156
160 void addUnlinkedNode(const std::string& nodeName)
161 {
162 m_unlinkedNodes.insert(nodeName);
163 }
164
173 virtual void integrityCheck() { }
174
185 virtual bool requireAllFields() const { return true; }
186
187protected:
195 Base(const DocumentWeakPtr& document, Cmper partId, ShareMode shareMode)
196 : m_document(document), m_partId(partId), m_shareMode(shareMode) {}
197
199 Base& operator=(const Base&) { return *this; }
200
201private:
202 const DocumentWeakPtr m_document;
203 const Cmper m_partId;
204 const ShareMode m_shareMode;
205 SharedNodes m_unlinkedNodes;
206};
