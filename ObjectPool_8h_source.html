49template <typename ObjectBaseType, typename TopKeyElementType = std::string>
50class ObjectPool
51{
52public:
54 using ObjectPtr = std::shared_ptr<ObjectBaseType>;
56 struct ObjectKey {
57 TopKeyElementType nodeId;
58 Cmper partId;
59 std::optional<Cmper> cmper1;
60 std::optional<Cmper> cmper2;
61 std::optional<Inci> inci;
62
64 ObjectKey(const TopKeyElementType n,
65 Cmper p,
66 std::optional<Cmper> c1 = std::nullopt,
67 std::optional<Cmper> c2 = std::nullopt,
68 std::optional<Inci> i = std::nullopt) : nodeId(n), partId(p), cmper1(c1), cmper2(c2), inci(i)
69 {
70 }
71
73 bool operator<(const ObjectKey& other) const
74 {
75 if (nodeId != other.nodeId) {
76 return nodeId < other.nodeId;
77 }
78 if (partId != other.partId) {
79 return partId < other.partId;
80 }
81 if (cmper1 != other.cmper1) {
82 return cmper1 < other.cmper1;
83 }
84 if (cmper2 != other.cmper2) {
85 return cmper2 < other.cmper2;
86 }
87 return inci < other.inci;
88 }
89 };
90
92 virtual ~ObjectPool() = default;

100 void add(const ObjectKey& key, ObjectPtr object)
101 {
102 m_pool.emplace(key, object);
103 }

117 template <typename T>
118 std::vector<std::shared_ptr<T>> getArray(const ObjectKey& key) const
119 {
120 std::vector<std::shared_ptr<T>> result;
121
122 auto rangeStart = m_pool.lower_bound(key);
123 auto rangeEnd = m_pool.upper_bound(
124 ObjectKey{
125 key.nodeId,
126 key.partId,
127 key.cmper1.value_or((std::numeric_limits<Cmper>::max)()),
128 key.cmper2.value_or((std::numeric_limits<Cmper>::max)()),
129 key.inci.value_or((std::numeric_limits<Inci>::max)())
130 }
131 );
132
133 for (auto it = rangeStart; it != rangeEnd; ++it) {
134 auto typedPtr = std::dynamic_pointer_cast<T>(it->second);
135 assert(typedPtr);
136 result.push_back(typedPtr);
137 }
138 return result;
139 }
140
152 template <typename T>
153 std::shared_ptr<T> get(const ObjectKey& key) const
154 {
155 auto it = m_pool.find(key);
156 if (it == m_pool.end()) {
157 return nullptr;
158 }
159 auto typedPtr = std::dynamic_pointer_cast<T>(it->second);
160 assert(typedPtr); // There is a program bug if the pointer cast fails.
161 return typedPtr;
162 }
163
174 template <typename T>
175 std::shared_ptr<T> getEffectiveForPart(const ObjectKey& key) const
176 {
177 if (auto partVersion = get<T>(key)) {
178 return partVersion;
179 }
180 if (key.partId == 0) {
181 // if this is already the score version, there is nothing to return.
182 return nullptr;
183 }
184 ObjectKey scoreKey(key);
185 scoreKey.partId = 0;
186 return get<T>(scoreKey);
187 }
188
189protected:
190 // prevent standalone construction
191 ObjectPool() = default;
192
193private:
194 std::map<ObjectKey, ObjectPtr> m_pool;
195};
