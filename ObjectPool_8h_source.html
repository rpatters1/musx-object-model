50template <typename ObjectBaseType, typename TopKeyElementType = std::string>
51class ObjectPool
52{
53public:
55 using ObjectPtr = std::shared_ptr<ObjectBaseType>;
57 struct ObjectKey {
58 TopKeyElementType nodeId;
59 Cmper partId;
60 std::optional<Cmper> cmper1;
61 std::optional<Cmper> cmper2;
62 std::optional<Inci> inci;
63
65 ObjectKey(const TopKeyElementType n,
66 Cmper p,
67 std::optional<Cmper> c1 = std::nullopt,
68 std::optional<Cmper> c2 = std::nullopt,
69 std::optional<Inci> i = std::nullopt) : nodeId(n), partId(p), cmper1(c1), cmper2(c2), inci(i)
70 {
71 }
72
74 bool operator<(const ObjectKey& other) const
75 {
76 if (nodeId != other.nodeId) {
77 return nodeId < other.nodeId;
78 }
79 if (partId != other.partId) {
80 return partId < other.partId;
81 }
82 if (cmper1 != other.cmper1) {
83 return cmper1 < other.cmper1;
84 }
85 if (cmper2 != other.cmper2) {
86 return cmper2 < other.cmper2;
87 }
88 return inci < other.inci;
89 }
90
92 std::string nodeString() const
93 {
94 if constexpr (std::is_same_v<TopKeyElementType, std::string>) {
95 return nodeId;
96 } else {
97 return std::string("entry ") + std::to_string(nodeId);
98 }
99 }
100 };
101
103 virtual ~ObjectPool() = default;
104
111 void add(const ObjectKey& key, ObjectPtr object)
112 {
113 m_pool.emplace(key, object);
114 auto it = m_shareMode.find(key.nodeId);
115 if (it == m_shareMode.end()) {
116 m_shareMode.emplace(key.nodeId, object->getShareMode());
117 } else if (object->getShareMode() != it->second && object->getShareMode() != Base::ShareMode::All) {
118 if (it->second == Base::ShareMode::All) {
119 m_shareMode[key.nodeId] = object->getShareMode();
120 } else {
121 MUSX_INTEGRITY_ERROR("Share mode for added " + key.nodeString() + " object [" + std::to_string(int(object->getShareMode()))
122 + "] does not match previous [" + std::to_string(int(it->second)) + "]");
123 }
124 }
125 }
126
139 template <typename T>
140 std::vector<std::shared_ptr<T>> getArray(const ObjectKey& key) const
141 {
142 std::vector<std::shared_ptr<T>> result;
143
144 auto rangeStart = m_pool.lower_bound(key);
145 auto rangeEnd = m_pool.upper_bound(
146 ObjectKey{
147 key.nodeId,
148 key.partId,
149 key.cmper1.value_or((std::numeric_limits<Cmper>::max)()),
150 key.cmper2.value_or((std::numeric_limits<Cmper>::max)()),
151 key.inci.value_or((std::numeric_limits<Inci>::max)())
152 }
153 );
154
155 for (auto it = rangeStart; it != rangeEnd; ++it) {
156 auto typedPtr = std::dynamic_pointer_cast<T>(it->second);
157 assert(typedPtr);
158 result.push_back(typedPtr);
159 }
160 return result;
161 }
162
175 template <typename T>
176 std::vector<std::shared_ptr<T>> getArrayForPart(const ObjectKey& key) const
177 {
178 if (key.partId == SCORE_PARTID) {
179 return getArray<T>(key);
180 }
181 auto it = m_shareMode.find(key.nodeId);
182 if (it == m_shareMode.end()) {
183 throw std::invalid_argument("Share mode not found for node " + key.nodeString());
184 }
185 switch (it->second) {
186 default:
187 case Base::ShareMode::All: {
188 ObjectKey scoreKey(key);
189 scoreKey.partId = SCORE_PARTID;
190 return getArray<T>(scoreKey);
191 }
192 case Base::ShareMode::None:
193 return getArray<T>(key);
194 case Base::ShareMode::Partial:
195 // it appears that all partially shareable nodes are created for each part, even if there is nothing in them.
196 // that means we will create an entire copy of the score values as needed, so here we can just return them.
197 return getArray<T>(key);
198 }
199 }
200
212 template <typename T>
213 std::shared_ptr<T> get(const ObjectKey& key) const
214 {
215 auto it = m_pool.find(key);
216 if (it == m_pool.end()) {
217 return nullptr;
218 }
219 auto typedPtr = std::dynamic_pointer_cast<T>(it->second);
220 assert(typedPtr); // There is a program bug if the pointer cast fails.
221 return typedPtr;
222 }
223
234 template <typename T>
235 std::shared_ptr<T> getEffectiveForPart(const ObjectKey& key) const
236 {
237 if (auto partVersion = get<T>(key)) {
238 return partVersion;
239 }
240 if (key.partId == SCORE_PARTID) {
241 // if this is already the score version, there is nothing to return.
242 return nullptr;
243 }
244 ObjectKey scoreKey(key);
245 scoreKey.partId = SCORE_PARTID;
246 return get<T>(scoreKey);
247 }
248
249protected:
250 // prevent standalone construction
251 ObjectPool() = default;
252
253private:
254 std::map<ObjectKey, ObjectPtr> m_pool;
255 std::unordered_map<TopKeyElementType, dom::Base::ShareMode> m_shareMode;
256};
